1. mysql -u root -p   //맨 처음 DB(mysql) 접속

2. 비밀번호 입력 (거의 '1234'로 되어 있을거 같음)

3. use SpringBootTest; //사용자 입력

4. 만약에 위의 글을 넣었을때 없다고 뜨거나 접속실패가 뜨면 사용자를 아직 만들지 않은거임.(5번부터 7번까지 참고해서 만들기.)

-- 5. create database nunnop2; //db 껍데기 제작

-- 6. grant all privileges on nunnop2.* to nunnop2@localhost identified by '1234';  // DB제작에 들어가는 기본 정보 입력

-- 7. flush privileges; //데이터베이스에서 사용자 권한 변경 사항을 즉시 적용하는 명령어 (여기까지가 제작 끝.)

8. use nunnop2; // 다시 입력해서 잘 만들어졌는지 확인.

9. show tables; // 테이블 보이기. (테이블 제작은 springBoot Entity에서 되어 있어야함.)
